package org.tcs;

import java.util.Scanner;

//find Even and odd no
public class Sample {
	public static void main(String[] args) {

		System.out.println("Please enter a no: ");
		Scanner s = new Scanner(System.in);
		int n1 = s.nextInt();
	

		if (n1%2==0) {
			System.out.println("Even No");
		} else {
			System.out.println("Odd No");
		}
	}

}
